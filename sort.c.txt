#include "datatype.h"

// Hoán đổi dữ liệu của 2 node
void swapData(Node* a, Node* b) {
    BaiTap temp = a->data;
    a->data = b->data;
    b->data = temp;
}

// Selection Sort theo độ khó
void selectionSortByDifficulty(Node* head) {
    for (Node* i = head; i != NULL; i = i->next) {
        Node* min = i;
        for (Node* j = i->next; j != NULL; j = j->next) {
            if (j->data.difficulty < min->data.difficulty) {
                min = j;
            }
        }
        if (min != i) {
            swapData(i, min);
        }
    }
}
